int fast_mul(int a, int b, int mod) {
	int ans = 0;
	while (b) {
		if (b & 1) {
			ans = (ans + a) % mod;
		}
		a = (a <<= 1) % mod;
		b >>= 1;
	}
	return ans;
}

int pow_mod(int a, int b, int mod) {
	int res = 1, base = a % mod;
	while (b) {
		if (b & 1) {
			res = fast_mul(res, base, mod);
		}
		base = fast_mul(base, base, mod);
		b >>= 1;
	}
	return res;
}

const int N = 1000010;
int prim[N];//存素数
int cnt=0;//质数个数
bool vis[N];//记录被划掉的合数

void get_prim(int x){
    for(int i=2;i<=x;i++){
        if(!vis[i])prim[++cnt]=i;
        for(int j=1;prim[j]<=x/i;j++){//越界中断
            vis[i*prim[j]]=true;
            if(i%prim[j]==0) break;//整除中断
        }
    }
}
